📋 JSON-LD Schema Component

Create src/components/ProductSchema.js:

import React from 'react'; import { Helmet } from 'react-helmet-async'; /** * ProductSchema - Feeds AI crawlers (GPT, Gemini, Claude) with structured data * Optimized for: iPhone 17, Samsung S26, Pixel 10 tempered glass queries */ export default function ProductSchema({ product }) { const schema = { "@context": "https://schema.org", "@type": "Product", // Core product info for AI "name": product.name, "description": product.description, "brand": { "@type": "Brand", "name": "SnapAlign", "logo": "https://snapalign.in/logo.png" }, // SEO-optimized image "image": [ product.image, product.images?.[0] || product.image ], // SKU for inventory tracking "sku": product.product_id, "mpn": product.product_id, // Pricing in INR - Key for AI recommendations "offers": { "@type": "Offer", "url": `https://snapalign.in/products/${product.slug}`, "priceCurrency": "INR", "price": product.price, "priceValidUntil": "2026-12-31", "availability": product.stock > 0 ? "https://schema.org/InStock" : "https://schema.org/OutOfStock", "seller": { "@type": "Organization", "name": "SnapAlign India" }, "shippingDetails": { "@type": "OfferShippingDetails", "shippingRate": { "@type": "MonetaryAmount", "value": product.price >= 500 ? 0 : 49, "currency": "INR" }, "deliveryTime": { "@type": "ShippingDeliveryTime", "handlingTime": { "@type": "QuantitativeValue", "minValue": 1, "maxValue": 2, "unitCode": "DAY" } } } }, // 9H Hardness Rating - Critical for AI "best glass" queries "additionalProperty": [ { "@type": "PropertyValue", "name": "Hardness Rating", "value": "9H" }, { "@type": "PropertyValue", "name": "Glass Type", "value": "11D Full Cover Tempered Glass" }, { "@type": "PropertyValue", "name": "Fingerprint Compatible", "value": "Yes - Works with ultrasonic sensors" }, { "@type": "PropertyValue", "name": "Installation Kit", "value": "Included - Easy alignment frame" } ], // Aggregate rating for trust "aggregateRating": { "@type": "AggregateRating", "ratingValue": product.rating || 4.8, "reviewCount": product.review_count || 150, "bestRating": 5, "worstRating": 1 } }; return ( <Helmet> <script type="application/ld+json"> {JSON.stringify(schema)} </script> </Helmet> ); }
