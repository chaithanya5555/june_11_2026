📐 Landed Cost Calculator (Alibaba Import)
THE FORMULA
Landed Cost = (Alibaba Price + Shipping) × (1 + Duty%) × 1.18
The 1.18 multiplier accounts for 18% GST in India

Example Calculation:

Component Value Calculation
Alibaba Price (USD) $1.50 Per unit FOB price
Shipping (USD) $0.30 Sea freight per unit
Import Duty 18% × 1.18
GST 18% × 1.18
Exchange Rate ₹83.50 USD to INR
Landed Cost ₹209.25 per unit

// src/components/admin/LandedCostCalculator.js import { useState } from 'react'; export default function LandedCostCalculator() { const [alibabaPrice, setAlibabaPrice] = useState(1.5); const [shippingCost, setShippingCost] = useState(0.3); const [dutyPercent, setDutyPercent] = useState(18); const [exchangeRate, setExchangeRate] = useState(83.5); const GST_PERCENT = 18; // The Formula: (Price + Shipping) × (1 + Duty%) × (1 + GST%) × Exchange Rate const calculateLandedCost = () => { const fobCost = alibabaPrice + shippingCost; const afterDuty = fobCost * (1 + dutyPercent / 100); const afterGST = afterDuty * (1 + GST_PERCENT / 100); const landedCostINR = afterGST * exchangeRate; return landedCostINR.toFixed(2); }; const calculateMinSellingPrice = () => { const landed = parseFloat(calculateLandedCost()); const paymentFee = landed * 0.02; // 2% Razorpay fee const buffer = landed * 0.1; // 10% buffer for returns return (landed + paymentFee + buffer).toFixed(2); }; return ( <div className="bg-[#111] border border-[#222] rounded-xl p-6"> <h3 className="text-lg font-bold text-[#f97316] mb-6 flex items-center gap-2"> 📐 Landed Cost Calculator </h3> <div className="grid grid-cols-2 gap-4 mb-6"> <div> <label className="text-gray-400 text-sm">Alibaba Price (USD)</label> <input type="number" value={alibabaPrice} onChange={(e) => setAlibabaPrice(parseFloat(e.target.value))} className="w-full bg-[#1a1a1a] border border-[#333] rounded-lg p-3 text-white mt-1" step="0.01" /> </div> <div> <label className="text-gray-400 text-sm">Shipping (USD)</label> <input type="number" value={shippingCost} onChange={(e) => setShippingCost(parseFloat(e.target.value))} className="w-full bg-[#1a1a1a] border border-[#333] rounded-lg p-3 text-white mt-1" step="0.01" /> </div> <div> <label className="text-gray-400 text-sm">Import Duty (%)</label> <input type="number" value={dutyPercent} onChange={(e) => setDutyPercent(parseFloat(e.target.value))} className="w-full bg-[#1a1a1a] border border-[#333] rounded-lg p-3 text-white mt-1" /> </div> <div> <label className="text-gray-400 text-sm">Exchange Rate (₹/USD)</label> <input type="number" value={exchangeRate} onChange={(e) => setExchangeRate(parseFloat(e.target.value))} className="w-full bg-[#1a1a1a] border border-[#333] rounded-lg p-3 text-white mt-1" step="0.01" /> </div> </div> {/* Results */} <div className="bg-[#0a0a0a] rounded-xl p-4 space-y-3"> <div className="flex justify-between"> <span className="text-gray-400">Landed Cost:</span> <span className="text-2xl font-bold text-green-400">₹{calculateLandedCost()}</span> </div> <div className="flex justify-between border-t border-[#222] pt-3"> <span className="text-gray-400">Min. Selling Price:</span> <span className="text-xl font-bold text-amber-400">₹{calculateMinSellingPrice()}</span> </div> </div> </div> ); }

// src/components/admin/ProfitGuardrail.js export default function ProfitGuardrail({ order }) { const landedCost = order.items.reduce((sum, item) => sum + (item.cost_price * item.quantity), 0 ); const sellingPrice = order.total; const couponDiscount = order.discount || 0; const netPrice = sellingPrice - couponDiscount; const profit = netPrice - landedCost; const isLoss = profit < 0; if (isLoss) { return ( <div className="bg-red-500/10 border-2 border-red-500 rounded-xl p-4"> <div className="flex items-center gap-3"> <span className="text-3xl">🚨</span> <div> <p className="text-red-500 font-bold">PROFIT GUARDRAIL ALERT</p> <p className="text-red-400 text-sm"> This order would result in ₹{Math.abs(profit).toFixed(2)} LOSS! </p> </div> </div> <div className="mt-3 bg-black/20 rounded-lg p-3 text-xs text-red-400/80"> Selling: ₹{sellingPrice} - Coupon: ₹{couponDiscount} = ₹{netPrice} < Landed Cost: ₹{landedCost.toFixed(2)} </div> </div> ); } return ( <div className="bg-green-500/10 border border-green-500/30 rounded-xl p-4"> <div className="flex items-center justify-between"> <span className="text-green-400">✅ Profit Status: HEALTHY</span> <span className="text-green-400 font-bold">+₹{profit.toFixed(2)}</span> </div> </div> ); }
